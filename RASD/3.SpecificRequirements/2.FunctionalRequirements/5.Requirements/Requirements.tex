\subsection{Requirements\label{sect:3.2.5}}
\begin{itemize}
	
	% TODO: 12a e 12b, aggiungere ultimi Req
	
	\item[R\subs{1}]At sign up, \ic{User} must provide: First name, last name, email and password;
	\item[R\subs{2}]At sign up, \ic{User} must accept \ic{Terms and conditions}, including the \ic{Privacy statement};
	\item[R\subs{3}]Identify a \ic{User} by his/her \ic{User identifier};
	\item[R\subs{4}]Receive \ic{User picture};
	\item[R\subs{5}]Receive \ic{User} choice for the type of \ic{Traffic violation};
	\item[R\subs{6}]Receive \ic{User position};
	\item[R\subs{7}]Send \ic{User picture} to the \ic{License plate recognition service};
	\item[R\subs{8}]Receive \ic{Ricognized license plate} from the \ic{License plate recognition service};
	\item[R\subs{9}]Create \ic{User report};
	\item[R\subs{10}]Store \ic{User report} in SafeStreets Database;
	\item[R\subs{11}]Send \ic{User report} to \ic{Municipality};
	\item[R\subs{12}]Store \ic{Ticket feedback} in SafeStreets Database;
	\item[R\subs{13}]Query SafeStreets Database for a \ic{User} by his/her \ic{User identifier};
	\item[R\subs{14}]Retrieve specific \ic{User reports} by querying SafeStreets Database;
	\item[R\subs{15}]Guarantees that the data which constitutes the \ic{User report} are not corrupted;
	\item[R\subs{16}]Receive \ic{Municipality} data access request;
	\item[R\subs{17}]Validate \ic{Municipality} data access request;
	\item[R\subs{18}]Allow \ic{Municipality} to set specific constraints to define \ic{Detailed statistics};
	\item[R\subs{19}]Send \ic{Detailed statistics} to the requesting \ic{Municipality};
	\item[R\subs{20}]Allow \ic{Users} to request all their \ic{User report} and the related \ic{Ticket feedback} stored in SafeStreets Database at any time;
	\item[R\subs{21}]Send to a specific \ic{User} his/her \ic{User report} stored;
	\item[R\subs{22}]Send to a specific \ic{User} the \ic{Ticket feedback} related to his/her \ic{User report};
	\item[R\subs{23}]Allow \ic{User} to choose among filters which define a group of \ic{User reports};
	\item[R\subs{24}]Send \ic{Public statistics} to the requesting \ic{User};
	\item[R\subs{25}]Send \ic{Accidents} Database access request to \ic{Municipality}; 
	\item[R\subs{26}]Receive \ic{Accidents} Database access confirmation from \ic{Municipality};
	\item[R\subs{27}]Request \ic{Accidents} data to \ic{Municipality};
	\item[R\subs{28}]Receive \ic{Accidents} data from \ic{Municipality};
	\item[R\subs{29}]Generate \ic{Possible interventions} crossing \ic{Municipality} \ic{Accidents} data with SafeStreets Database data.
	\item[R\subs{30}]Send generated \ic{Possible interventions} to \ic{Municipality};
	\item[R\subs{31}]At sign up, \ic{Municipality} must provide a \ic{Reference code};
	\item[R\subs{32}]Identify \ic{Municipality} by its \ic{Reference code};
	\item[R\subs{33}]Receive \ic{Ticket feedback} from \ic{Municipality};

	
\end{itemize}
